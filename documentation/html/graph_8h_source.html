414 template<typename E>
415 void Graph<E>::Dijkstra(const E &e)
416 {
417  // Comparitor class for vertexItr, pushes smallest weights to the bottom
418  struct distanceCompare {
419  bool operator()(const VertexItr &lhs, const VertexItr &rhs) {
420  return (lhs->getDistance() > rhs->getDistance());
421  }
422  } compareDist;
423 
424  // Reset the weight and parent of all vertex
425  resetDijkstra();
426  // find the vertex to use as a starting point
427  VertexItr eItr = findVertex(e);
428 
429  // set source distance to 0
430  eItr->setDistance(0);
431 
432  // Load iterators to vertices into a psudo PQueue
433  VertexItrVector unusedVertex;
434  for(VertexItr i = vertices_.begin(); i != vertices_.end(); i++){
435  unusedVertex.push_back(i);
436  }
437 
438  // actual algorithm
439  while(!unusedVertex.empty()){
440  // re-sort every time since weights have changed
441  std::sort(unusedVertex.begin(), unusedVertex.end(), compareDist);
442 
443  // grab the next vertex with the smallest distance, and then pop it
444  VertexItr currentVertex = unusedVertex.back();
445  unusedVertex.pop_back();
446 
447  // IF the smallest vertex has weight infintiy then it is unconnected, skip
448  if( currentVertex->getDistance() != INF ){
449  // get iterators to all the neightbors of current vertex
450  VertexItrVector neighbors = currentVertex->adjacentVertex();
451  for(typename VertexItrVector::iterator j = neighbors.begin(); j != neighbors.end(); j++){
452  // relaxation algorithm
453  int sumDistance = currentVertex->getDistance() + currentVertex->distanceTo(*j);
454  if( sumDistance < (**j).getDistance() ){
455  (**j).setDistance(sumDistance);
456  (**j).setParent(currentVertex);
457  }
458  }//END FOR LOOP
459  }//END INF IF STATEMENT
460  }//END WHILE
461 }//END DIJKSTRA
