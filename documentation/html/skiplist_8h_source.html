395 template <typename K, typename V>
396 typename skiplist<K,V>::node *skiplist<K,V>::search(const K &k) const {
397  node tempNode;
398  node* itr = head_;
399  tempNode.add(item(k, V())); // Created a dumy entry for searching
400 
401  while(itr->down() != nullptr) {
402  if (tempNode == *itr->right() ) {
403  // Move right if equal
404  itr = itr->right();
405  }
406  else if (tempNode < *itr->right() ) {
407  // Move down if less than
408  itr = itr->down();
409  }
410  else if (tempNode > *itr->right() ) {
411  // Move right if greater than
412  itr = itr->right();
413  }
414  }
415 
416  // If the bottom is encountered before we get to the desired position, the iterator
417  // Needs to move along the bottom till it gets to the right position for RIGHT insert
418  while(itr->right()->type() != node::TAIL && tempNode >= *itr->right()) {
419  itr = itr->right();
420  }
421  return itr;
422 }

429 void skiplist<K,V>::erase(const K &k) {
430  node* itr = search(k);
431 
432  // If the key of the found location matches the key we want to delete then do the dirty
433  if(itr->key() == k) {
434  while(itr != nullptr) {
435  // If there is a node to the left, link it to the right
436  if (itr->left() != nullptr) {
437  itr->left()->setRight(itr->right());
438  }
439  // If there is a node to the right, link it to the left
440  if (itr->right() != nullptr) {
441  itr->right()->setLeft(itr->left());
442  }
443  // Move up in the stack
444  node *tempItr = itr->up();
445  delete itr;
446  itr = tempItr;
447  }// END OF WHILE LOOP
448  size_--;
449  }// END OF IF STATEMENT

456 template <typename K, typename V>
457 void skiplist<K,V>::insert(const item &e) {
458  node* newNode = new node;
459  node* location;
460  newNode->add(e);
461  location = search(e.key());
462 
463  // If the key already exists, add the new item to each node in the tower
464  if(*location == *newNode){
465  while(location != nullptr) {
466  location->add(e);
467  location = location->up();
468  }
469  }
470  // If the key did not exist in the table add it and create a tower of random height
471  else {
472  // Insert the new node to the RIGHT of the position found
473  // First link the new node to its neighbors
474  newNode->setLeft(location);
475  newNode->setRight(location->right());
476  // Then the neighbors to the new node
477  newNode->left()->setRight(newNode);
478  newNode->right()->setLeft(newNode);
479 
480  node* leftScan = location; // Pointer for scanning left to find a horizontal links
481  int level = 1; // Counter to keep track of the current level
482 
483  // Make a new column of a random height by flipping a "coin" repeatedly
484  while(flipCoin()) {
485  // If the top level is reached, make a new one!
486  if(level >= height_) { addBlankLevel(); }
487 
488  // Scan left for a node with an up-pointer
489  while(leftScan->up() == nullptr) {
490  leftScan = leftScan->left();
491  }
492  // Move the scanner up one
493  leftScan = leftScan->up();
494 
495  // insert the new node above the previous level, with data
496  newNode->setUp(new node(nullptr, newNode, leftScan, leftScan->right()));
497  newNode->up()->add(e);
498  // Connect the neightbors to the new node
499  leftScan->right()->setLeft(newNode->up());
500  leftScan->setRight(newNode->up());
501 
502  //Prepare for next iteration if there is one
503  newNode = newNode->up();
504  level++;
505 
506  }//END OF TOWER BUILDING WHILE LOOP
507  //increment the size of the skpi list
508  size_++;
509  }//END OF NEW NODE INSERT
510 }//END OF INSERT METHOD
